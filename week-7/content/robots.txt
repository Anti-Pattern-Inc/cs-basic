    const mainVisual = document.getElementById('mainVisual');
    const mainVisualHeight = mainVisual.offsetHeight;
    const nav = document.getElementById('nav')
    const navHeight = nav.offsetHeight;
    const toggleButton = document.getElementById('toggleButton')
    let hamburgerIsOpen = false;
    toggleButton.addEventListener('click', () => {
      let scrollPosition = window.scrollY;
      let isAboveMainVisual = navHeight + scrollPosition <= mainVisualHeight;
      // ハンバーガーが開いているかどうかのBooleanを反転
      hamburgerIsOpen = hamburgerIsOpen ? false : true
      if(hamburgerIsOpen) {
        // ハンバーガーが開いたらbg-navyにする
        nav.classList.add('bg-navy')
        return
      } 
      if(!hamburgerIsOpen && isAboveMainVisual) {
        // ハンバーガーが閉じるとき、かつ、MainVisualより上にあるときはbg-navyを外す
        nav.classList.remove('bg-navy')
        return
      }
    })
    window.addEventListener('scroll', () => {
      let scrollPosition = window.scrollY;
      let isAboveMainVisual = navHeight + scrollPosition <= mainVisualHeight;
      if (hamburgerIsOpen) {
        // ハンバーガーが開いていたら、bg-navyのままで何もしない
        return
      }
      if (isAboveMainVisual) {
        // MainVisualより上だったらbg-navyを外す
        nav.classList.remove('bg-navy')
        return
      }
      if (!isAboveMainVisual) {
        // MainVisualより下だったらbg-navyにする
        nav.classList.add('bg-navy')
        return
      }
    })
